class Propagator(){

    private:
        vectorf<void(fn)(void)>
        vectorp<void(fn)(void)>

    public:
        int agregarPerturbacion(void(fn)(void), void* param){
            vectorf.push_back(fn)
            vectorp.push_back(param)
        }

        void propagate(){
            i=0;
            for f in vectorf:
                 f(vectorp[i])
                 i++
        }

};
